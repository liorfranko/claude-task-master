Extended configuration schema:
```javascript
// Extension to existing config schema
const mondayConfigSchema = {
  mondayIntegration: {
    boardId: { type: 'string', required: true },
    apiToken: { type: 'string', required: false }, // Can be provided via env var
    columnMapping: {
      status: { type: 'string', default: 'status' },
      title: { type: 'string', default: 'name' },
      description: { type: 'string', default: 'notes' }
    },
    syncSettings: {
      autoSync: { type: 'boolean', default: false },
      syncSubtasks: { type: 'boolean', default: false }
    }
  }
};

// Function to validate Monday.com configuration
async function validateMondayConfig(config, mondayClient) {
  // Check if board is accessible
  const query = `
    query {
      boards(ids: [${config.mondayIntegration.boardId}]) {
        id
        name
        columns {
          id
          title
          type
        }
      }
    }
  `;
  
  try {
    const result = await mondayClient._executeWithRateLimit(query);
    if (!result.boards || result.boards.length === 0) {
      return { valid: false, error: 'Board not found or not accessible' };
    }
    return { valid: true, board: result.boards[0] };
  } catch (error) {
    return { valid: false, error: error.message };
  }
}

// Function to get API token (from config or env var)
function getMondayApiToken(config) {
  return config.mondayIntegration?.apiToken || process.env.MONDAY_API_TOKEN;
}
```
