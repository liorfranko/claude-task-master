```javascript
// In commands.js
const mondaySyncEngine = require('./modules/monday-sync');

// Add to commands object
const commands = {
  // ... existing commands
  'update-monday-from-local': updateMondayFromLocal,
  'sync': handleSyncCommands
};

async function updateMondayFromLocal(args) {
  const options = {
    taskId: args['--task-id'],
    dryRun: args['--dry-run'] === true,
    force: args['--force'] === true
  };
  
  console.log('Pushing tasks to Monday.com...');
  
  if (options.dryRun) {
    console.log('DRY RUN: No changes will be made to Monday.com');
  }
  
  if (options.taskId) {
    console.log(`Pushing task ID: ${options.taskId}`);
  } else {
    console.log('Pushing all tasks');
  }
  
  try {
    const results = await mondaySyncEngine.pushAllTasksToMonday(options);
    
    console.log('\nSync Results:');
    for (const result of results) {
      if (result.success) {
        console.log(`✅ Task ${result.taskId}: Successfully synced to Monday.com${result.dryRun ? ' (dry run)' : ''}`);
      } else {
        console.log(`❌ Task ${result.taskId}: Failed to sync - ${result.error}`);
      }
    }
    
    console.log('\nSync completed.');
  } catch (error) {
    console.error('Error during sync:', error.message);
  }
}

async function handleSyncCommands(args) {
  const subcommand = args._[1];
  
  switch (subcommand) {
    case 'push-to-monday':
      return updateMondayFromLocal(args);
    // Other sync subcommands will be added later
    default:
      console.error(`Unknown sync subcommand: ${subcommand}`);
      console.log('Available subcommands: push-to-monday');
  }
}
```
