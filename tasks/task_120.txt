Implement CLI commands:
```javascript
// In commands.js
function configureMonday(args) {
  const config = loadConfig();
  
  if (args['--monday-board']) {
    config.persistence = config.persistence || {};
    config.persistence.mondayConfig = config.persistence.mondayConfig || {};
    config.persistence.mondayConfig.boardId = args['--monday-board'];
  }
  
  if (args['--monday-token']) {
    config.persistence = config.persistence || {};
    config.persistence.mondayConfig = config.persistence.mondayConfig || {};
    config.persistence.mondayConfig.apiToken = args['--monday-token'].startsWith('env:') 
      ? args['--monday-token'] 
      : `env:${args['--monday-token']}`;
  }
  
  if (args['--set-mode']) {
    const mode = args['--set-mode'];
    if (!['local', 'monday', 'hybrid'].includes(mode)) {
      console.error(`Invalid mode: ${mode}. Must be one of: local, monday, hybrid`);
      return;
    }
    config.persistence = config.persistence || {};
    config.persistence.mode = mode;
  }
  
  saveConfig(config);
  console.log('Monday.com configuration updated successfully');
}
```
